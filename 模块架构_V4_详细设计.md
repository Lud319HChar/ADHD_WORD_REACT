### 2.2 各场景路由详解

#### 训练场景 — handleK() 详解
```javascript
function handleK() {
    // 自动播放模式：K 作为"跳过"键
    if (state.trainMode === TRAIN_MODES.AUTO_PLAY && state.currentState !== STATE.IDLE) {
        invalidateSession();
        import('./core.js').then(m => m.skipToNext());  // 动态import跳过
        return;
    }
    // 手动模式：K 作为"认识"键
    if (state.currentState === STATE.PLAYING || state.currentState === STATE.WAIT_RESPONSE) {
        transition('K');
    }
}
```

#### 训练场景 — handleJ() 详解（双分支语义）
```javascript
function handleJ() {
    // 分支1：SHOW_ANSWER 状态下 → 直接标记不熟（用于A/D回退后的历史词标记）
    if (state.currentState === STATE.SHOW_ANSWER) {
        if (state.currentWord && !state.unfamiliarWords.includes(/*...*/)) {
            addToUnfamiliar(state.currentWord);
            state.unfamiliarCount++;
        }
        showUnfamiliarIndicator();  // 红色脉冲动画
        transition('SPACE');
        return;
    }
    // 分支2：PLAYING / WAIT_RESPONSE 状态下 → 正常不熟流程
    if (state.currentState === STATE.PLAYING || state.currentState === STATE.WAIT_RESPONSE) {
        transition('J');
        showUnfamiliarIndicator();
    }
}
```

### 2.3 词库渲染

**完整词库** (`renderWordLibrary(filterType)`)：
- 根据 `filterType` 过滤 `wordDB`（14 种：全部/5词性/6主题/3难度）
- 渲染为 `.word-card` 网格（每卡片：单词+多译标记/音标/释义/例句/标签行）
- 带 `MULTI_TRANSLATION_WORDS` 和 `MANY_TRANSLATION_WORDS` 的多译标记
- 词性颜色：名词蓝/动词绿/形容词橙/副词紫/介词粉

**不熟词库** (`renderUnfamiliarLibrary()`)：
- 从 `wordDB` 中筛选 `id ∈ state.unfamiliarWords` 且 `id < 10000` 的单词
- 每个词卡带「移除」按钮 + 导出/导入按钮
- 操作提示行："L=不熟训练 | WASD=移动 | J=移除"

## 三、引擎层 — core.js

### 3.1 状态转移函数 `transition(event)`

```javascript
export function transition(event) {
    // 输入锁检查（Space 在白名单内，总是允许暂停）
    if (state.isProcessingInput && event !== 'SPACE') return;

    const nextState = getNextState(state.currentState, event);
    if (state.currentState === nextState && event !== 'NEXT_WORD') return;

    state.isProcessingInput = true;
    const previousState = state.currentState;
    invalidateSession();  // playSessionID++ + 清定时器 + 取消语音

    switch (nextState) {
        case STATE.IDLE:          handleIdleState(); break;
        case STATE.PLAYING:       handlePlayingState(event, previousState); break;
        case STATE.WAIT_RESPONSE: handleWaitResponseState(event); break;
        case STATE.SHOW_ANSWER:   handleShowAnswerState(event); break;
    }
    updateState({ currentState: nextState });
}
```

### 3.2 各状态处理函数

#### handleIdleState()
```
清空舞台：显示"按SPACE开始" + 隐藏单词/音标/释义/例句
释放输入锁
```

#### handlePlayingState(event, previousState)
```
三个分支：
  1. prevState === WAIT_RESPONSE
     → playKnowSound() + updateMastery(+1) + updateCombo(combo+1) + nextWord()
  2. prevState === SHOW_ANSWER && AUTO_PLAY
     → nextWord()
  3. 首次进入（prevState 为 IDLE 或 SPACE）
     → getRandomWord() → displayWord() → addToHistory()
     → if AUTO_PLAY: speakAndAutoShow()
     → else if speed !== 5: speakWordAsync().then({if PLAYING → transition('SPEAK_END')})
     → else (speed=5): safeScheduleTimer(transition('SPEAK_END'), gap)
释放输入锁
```

#### speakAndAutoShow(word) — 自动播放模式专用
```
speakWordAsync(en, rate) →
  .then: showAnswer() + playShowMeaningSound() + updateState(SHOW_ANSWER)
  safeScheduleTimer(transition('NEXT_WORD'), gap)
```

#### nextWord()
```
safeScheduleTimer → 采样新词 → displayWord + addToHistory
  如果是 AUTO_PLAY → speakAndAutoShow
  否则 → speakWordAsync 或 safeScheduleTimer
```

#### handleWaitResponseState()
```
showAnswer() + playShowMeaningSound()
释放输入锁
```

#### handleShowAnswerState(event)
```
if event === 'J':
  playUnfamiliarSound()
  updateMastery(-2) + updateCombo(0)
  addToUnfamiliar() + state.unfamiliarCount++
showAnswer() + playShowMeaningSound()
safeScheduleTimer(transition('NEXT_WORD'), gap)  ← 超时自动下一词
释放输入锁
```

### 3.3 选词算法 `getRandomWord()`

```
Step 1 — 词库过滤：
  FULL       → wordDB 全部
  SUPER_HF   → wordDB ∩ SUPER_HF_WORDS
  PHRASES    → phraseDB → 映射为 {id: i+10000, word: phrase, meaning, example, category}
  KEYWORDS   → keywordsDB.wordIds 合并去重 → 对应 wordDB 子集

Step 2 — 可用词过滤：
  排除 recentWordsWindow 中的词（20词环形队列）
  排除 nextReviewTime 未到的词

Step 3 — 加权随机采样：
  for each word:
    BaseWeight = max(1, 11 - masteryScore)        [熟练度: -10→21, 10→1]
    TimeFactor = 1 + (now - lastSeen) / 24小时
    weight = BaseWeight × TimeFactor
  → 汇总 totalWeight → random落点选择

Step 4 — 防重注册：
  将选中词id推入 recentWordsWindow（超20则shift最早）
  返回选中词对象
```

### 3.5 熟练度系统

#### updateMastery(word, success)
```javascript
if (word.id >= 10000) return;                      // 短语不参与
const target = wordDB.find(w => w.id === word.id);

if (success) {  // K
    target.masteryScore = min(10, target.masteryScore + 1);
    target.nextReviewTime = now + masteryScore × 30min;
    state.knowCount++;
} else {         // J
    target.masteryScore = max(-10, target.masteryScore - 2);
    target.nextReviewTime = now + 90s;             // 紧急回流
    state.unfamiliarCount++;
}
target.appearCount++;
target.lastSeenTimeStamp = now;
saveProgress();  // 3秒节流写入
```

```javascript
removeFromUnfamiliar:
```

### 4.2 状态转移矩阵 `getNextState(currentState, event)`

```javascript
const transitions = {
```

### 4.3 per-mode 历史栈操作

```javascript
const HISTORY_MAX = 16;

function currentHistory() {
    return state.historyByMode[state.trainMode];
}

function addToHistory(wordId) {
    const h = currentHistory();
    if (h.pointer === h.stack.length - 1) {
        // 在栈顶 → push + shift(超16)
        h.stack.push(wordId);
        if (h.stack.length > HISTORY_MAX) h.stack.shift();
        h.pointer = h.stack.length - 1;
    } else {
        // 不在栈顶（用户A键回退后又走新词）→ 截断旧分支 + 新建分支
        h.stack = h.stack.slice(0, h.pointer + 1);
        h.stack.push(wordId);
        if (h.stack.length > HISTORY_MAX) h.stack = h.stack.slice(-HISTORY_MAX);
        h.pointer = h.stack.length - 1;
    }
    window.dispatchEvent(new CustomEvent('word-added'));
}

function goPrevWord() {
    const h = currentHistory();
    if (h.pointer > 0) { h.pointer--; return h.stack[h.pointer]; }
    return null;
}

function goNextWord() {
    const h = currentHistory();
    if (h.pointer < h.stack.length - 1) { h.pointer++; return h.stack[h.pointer]; }
    return null;
}
```

### 5.1 Web Audio API 音效合成

```javascript
let audioContext = null;  // 懒加载

function getAudioContext() {
    if (!audioContext)
        audioContext = new (window.AudioContext || window.webkitAudioContext)();
    return audioContext;
}

export function playAudio(frequency, duration, type) {
    const ctx = getAudioContext();
    const osc = ctx.createOscillator();   // 振荡器
    const gain = ctx.createGain();        // 增益节点

    osc.connect(gain);
    gain.connect(ctx.destination);

    osc.type = type;
    osc.frequency.setValueAtTime(frequency, ctx.currentTime);

    gain.gain.setValueAtTime(0.3, ctx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + duration);
    // ↑ 指数衰减淡出，避免突兀切断

    osc.start(ctx.currentTime);
    osc.stop(ctx.currentTime + duration);
}
```

### 5.4 Web Speech API TTS （异步版 — 带会话锁，训练核心）

```javascript
export function speakWordAsync(text, lang, rate = 1) {
    return new Promise((resolve, reject) => {
        if (!('speechSynthesis' in window)) {
            resolve({ status: "not_supported" });
            return;
        }

        const capturedSessionID = state.playSessionID;  // 闭包死锁

        window.globalAudioGarbageCollectorBucket = null;  // 防止GC回收

        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = lang;
        utterance.rate = rate;

        window.globalAudioGarbageCollectorBucket = utterance;  // 挂载全局防GC

        utterance.onend = (event) => {
            if (state.playSessionID !== capturedSessionID) {
                resolve({ status: "aborted" });  // 会话过期 → 熔断
                return;
            }
            resolve({ status: "success" });
        };

        utterance.onerror = (event) => {
            console.warn("TTS Exception:", event.error);
            resolve({ status: "error" });  // 安全解构，不抛异常
        };

        window.speechSynthesis.speak(utterance);
    });
}
```
